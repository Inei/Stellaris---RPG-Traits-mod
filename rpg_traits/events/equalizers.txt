namespace = equalizer

planet_event = {
	id = equalizer.1
	hide_window = yes
	
	trigger = {
		any_pop = {
			AND = {
				has_trait = trait_titanic
				has_trait = trait_rapid_breeders1
			}
			is_being_purged = no
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		every_owned_pop = {
			limit = {
				AND = {
					has_trait = trait_titanic
					has_trait = trait_rapid_breeders1
				}
			}
			remove_modifier = titanic_grow1
			remove_modifier = titanic_grow2
			remove_modifier = titanic_grow3
			add_modifier = {
				modifier = titanic_grow1
				days = -1
			}
		}
	}
}
